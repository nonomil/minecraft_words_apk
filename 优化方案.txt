function lazyLoadImages() {
    const images = document.querySelectorAll('img[data-src]');
    images.forEach(img => {
        if (isInViewport(img)) {
            img.src = img.dataset.src;
            img.removeAttribute('data-src');
        }
    });
}

激活码系统：
javascriptasync function validateActivationCode(code) {
    try {
        const response = await fetch('https://raw.githubusercontent.com/nonomil/minecraft_words_apk/main/activation_codes.txt');
        const codes = await response.text();
        
        if (codes.includes(code.trim())) {
            localStorage.setItem('activation_code', code);
            localStorage.setItem('activated', 'true');
            alert('🎉 激活成功！开始学习吧！');
            return true;
        } else {
            alert('❌ 激活码无效，请联系客服');
            return false;
        }
    } catch (error) {
        alert('网络错误，请稍后重试');
        return false;
    }
}

javascriptconst achievements = {
    'first_word': { name: '第一个单词', icon: '🌟', desc: '学会第一个单词' },
    'speed_learner': { name: '学习达人', icon: '⚡', desc: '连续答对10个' },

function checkAchievements() {
    // 检查各种成就条件
    const learnedCount = JSON.parse(localStorage.getItem('learnedWords') || '[]').length;
    
    if (learnedCount >= 1 && !hasAchievement('first_word')) {
        showAchievement('first_word');
    }
}

function showAchievement(achievementId) {
    const achievement = achievements[achievementId];
    const notification = document.createElement('div');
    notification.className = 'achievement-popup';
    notification.innerHTML = `
        <div class="achievement-content">
            <span class="achievement-icon">${achievement.icon}</span>
            <div>
                <h4>🎉 获得成就！</h4>
                <p><strong>${achievement.name}</strong></p>
                <p>${achievement.desc}</p>
            </div>
        </div>
    `;
    document.body.appendChild(notification);
    
    setTimeout(() => notification.remove(), 3000);
}

javascript// 发送匿名数据帮助你了解用户使用情况
function trackEvent(event, data = {}) {
    // 可以发送到免费的分析服务
    fetch('https://your-analytics.com/track', {
        method: 'POST',
        body: JSON.stringify({
            event,
            data,
            timestamp: Date.now(),
            userAgent: navigator.userAgent.slice(0, 50)
        })
    }).catch(() => {}); // 静默失败，不影响用户体验
}

// 使用示例
trackEvent('word_learned', { word: currentWord });
trackEvent('test_completed', { score: correctCount });
