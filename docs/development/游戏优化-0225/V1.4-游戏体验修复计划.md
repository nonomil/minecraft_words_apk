### Step 2：B-1 村庄单词屋词库改为进村前接触过的单词

**问题根因：** `buildVillageChallengeWords` 只用 `getVillageWords(village.biomeId)` 取群系词库，与玩家实际接触过的词无关。

**修复方案：** 优先从 `sessionCollectedWords` 里取进村前接触过的词，不足时 fallback 到群系词库。

定位 `12-village-challenges.js` 的 `buildVillageChallengeWords`：

```js
function buildVillageChallengeWords(village, questionCount) {
    // 新增：优先使用进村前接触过的单词
    const sessionWords = Array.isArray(sessionCollectedWords)
        ? sessionCollectedWords
            .filter(w => w && w.en)
            .map(resolveVillageChallengeWord)
            .filter(w => !!w)
        : [];

    // 去重
    const seen = new Set();
    const uniqueSessionWords = sessionWords.filter(w => {
        if (seen.has(w.en)) return false;
        seen.add(w.en);
        return true;
    });

    // 打乱顺序，取前 questionCount 个
    const shuffled = uniqueSessionWords.sort(() => Math.random() - 0.5);
    if (shuffled.length >= questionCount) {
        return shuffled.slice(0, questionCount);
    }

    // 不足时 fallback 到群系词库补充
    const biomeWords = typeof getVillageWords === 'function'
        ? getVillageWords(village.biomeId) : [];
    const normalized = Array.isArray(biomeWords)
        ? biomeWords.map(resolveVillageChallengeWord).filter(w => !!w && !seen.has(w.en))
        : [];

    return [...shuffled, ...normalized.sort(() => Math.random() - 0.5)]
        .slice(0, questionCount);
}
```

**注意：** `sessionCollectedWords` 在 `01-config.js` 中声明为全局变量，`12-village-challenges.js` 可直接访问。

### Step 4：B-3 烈焰人援军形象改为小火球

**问题根因：** 援军当前是普通方块敌人，视觉上无法区分。

先定位援军生成代码：

```bash
grep -n "reinforc\|blaze\|援军\|summon" \
  apk/src/modules/13-game-loop.js \
  apk/src/modules/11-game-init.js
```

**修复方案：** 给援军敌人加 `type: 'blaze_mini'` 标识，渲染时画橙色圆形 + 火焰粒子：

```js
// 援军生成时
enemies.push({
    x: bossX + offset,
    y: bossY,
    type: 'blaze_mini',      // 新增标识
    width: 20, height: 20,   // 比普通敌人小
    hp: 2,
    damage: 1,
    color: '#FF6D00',
    // ...其他属性
});
```

渲染模块中为 `blaze_mini` 添加专属绘制逻辑：

```js
if (enemy.type === 'blaze_mini') {
    // 橙色火球主体
    ctx.beginPath();
    ctx.arc(cx, cy, enemy.width / 2, 0, Math.PI * 2);
    ctx.fillStyle = '#FF6D00';
    ctx.fill();
    // 内部亮核
    ctx.beginPath();
    ctx.arc(cx, cy, enemy.width / 4, 0, Math.PI * 2);
    ctx.fillStyle = '#FFEB3B';
    ctx.fill();
    // 火焰粒子（每帧随机偏移）
    for (let i = 0; i < 3; i++) {
        const px = cx + (Math.random() - 0.5) * enemy.width;
        const py = cy - enemy.height / 2 - Math.random() * 8;
        ctx.beginPath();
        ctx.arc(px, py, 2, 0, Math.PI * 2);
        ctx.fillStyle = `rgba(255, ${100 + Math.floor(Math.random()*100)}, 0, 0.8)`;
        ctx.fill();
    }
}
```

### Step 5：B-4 钻石盔甲消失后背包仍可装备的 Bug

**问题根因：** `breakEquippedArmor` 清空了 `playerEquipment.armor`，但如果 `inventory` 里还有该盔甲的残留条目，玩家可以再次装备。

定位 `equipArmorFromBackpack` 和 `breakEquippedArmor`，确认装备时是否从 inventory 移除：

```js
// breakEquippedArmor 末尾追加：清除 inventory 里的同类盔甲残留
function breakEquippedArmor(armorId) {
    playerEquipment.armor = null;
    playerEquipment.armorDurability = 0;
    playerEquipment.armorEquippedAt = 0;
    playerEquipment.armorLastDurabilityTick = 0;
    // 新增：同步清除 inventory 里的残留
    if (inventory && inventory[armorId] > 0) {
        inventory[armorId] = 0;
    }
    showToast(`${ARMOR_TYPES?.[armorId]?.name || '盔甲'} 已失效`);
    updateArmorUI();
    updateInventoryUI();  // 新增：刷新背包 UI
}
```

### Step 8：B-7 隐藏宝箱下方地面第一次挖不消失

**问题根因：** 挖地时地面方块直接移除，宝箱掉落后无法站立打开。

**修复方案：** 给隐藏宝箱下方的地面方块加 `hiddenChestFloor: true` 标记，挖掘逻辑检测此标记：

```js
// 生成隐藏宝箱时，标记下方地面
function spawnHiddenChest(x, groundY) {
    chests.push(new Chest(x, groundY - blockSize, true)); // hidden=true
    // 标记下方地面方块
    const floorPlatform = platforms.find(p =>
        p.x <= x && p.x + p.width >= x + blockSize &&
        Math.abs(p.y - groundY) < 4
    );
    if (floorPlatform) floorPlatform.hiddenChestFloor = true;
}

// 挖掘逻辑中
function digGroundBlock(platform) {
    if (platform.hiddenChestFloor) {
        // 检查上方宝箱是否已打开
        const chestAbove = chests.find(c =>
            c.hidden && Math.abs(c.x - platform.x) < blockSize &&
            Math.abs(c.y - platform.y - blockSize) < 4
        );
        if (chestAbove && !chestAbove.opened) {
            showToast('💡 先打开宝箱！');
            return; // 宝箱未开，地面不消失
        }
        platform.hiddenChestFloor = false; // 宝箱已开，正常挖
    }
    platform.remove = true;
}
```
