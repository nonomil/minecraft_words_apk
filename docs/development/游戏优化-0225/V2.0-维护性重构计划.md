### Step 2：架构-1 拆分 `damagePlayer`（`13-game-loop.js`）

将 ~230 行的超级函数拆为两层：

```js
// 层1：纯计算，无副作用，可单独测试
function calcActualDamage(baseDamage, armorId, durability, defense, difficulty) {
    const scaledDamage = baseDamage * (difficulty?.enemyDamageMult ?? 1.0);
    const reductionByArmor = {
        leather: 0.2, chainmail: 0.35, iron: 0.55,
        gold: 0.65, diamond: 1.0, netherite: 1.0
    };
    const hasArmor = armorId && durability > 0;
    const baseReduction = Math.min(0.6, defense * 0.12);
    const armorReduction = reductionByArmor[armorId];
    const reduction = hasArmor
        ? (Number.isFinite(armorReduction) ? armorReduction : baseReduction)
        : baseReduction;
    const minDamage = hasArmor ? 0 : 1;  // ← 若已确认 diamond 无敌为 bug，此处改为 1
    return Math.max(minDamage, Math.round(scaledDamage * (1 - reduction)));
}

// 层2：状态应用，调用层1
function damagePlayer(baseDamage, source) {
    const actualDamage = calcActualDamage(
        baseDamage,
        inventory.armor?.id,
        inventory.armor?.durability,
        player.defense ?? 0,
        currentDifficulty
    );
    // ... 击退、HP 扣减、UI 更新、游戏结束判断 ...
}
```

### Step 3：架构-2 `useInventoryItem` 改为 handler map（`13-game-loop.js`）

```js
const ITEM_HANDLERS = {
    gunpowder: (inv, player) => {
        inv.gunpowder--;
        bombs.push(createBomb(player.x, player.y, player.facingRight ? 1 : -1));
        showToast('💣 投掷炸弹');
    },
    diamond: (inv) => {
        if (playerHp >= playerMaxHp) { showToast('❤️ 已满血'); return false; }
        inv.diamond--;
        healPlayer(1);
        showToast('💎 恢复生命');
    },
    // ... 其余物品 handler ...
};

function useInventoryItem(itemKey) {
    const handler = ITEM_HANDLERS[itemKey];
    if (!handler) return;
    const result = handler(inventory, player);
    if (result !== false) {
        updateHpUI();
        updateInventoryUI();
    }
}
```

新增物品只需在 `ITEM_HANDLERS` 加一条，不需要动主函数。

### Step 4：性能-1 高频 filter 改为批量清理（`13-game-loop.js`）

定位每帧执行的数组清理：

```js
// 改前（每帧重建数组）
decorations = decorations.filter(d =>
    d.x + d.width > cameraX - removeThreshold && !d.remove);
enemies = enemies.filter(e => !e.remove && e.y < 950);

// 改后（每 60 帧清理一次）
if (gameFrame % 60 === 0) {
    decorations = decorations.filter(d =>
        !d.remove && d.x + d.width > cameraX - removeThreshold);
    enemies = enemies.filter(e => !e.remove && e.y < 950);
    items = items.filter(i => !i.remove);
    particles = particles.filter(p => !p.remove);
    floatingTexts = floatingTexts.filter(t => !t.remove);
}
```

注意：标记删除（`obj.remove = true`）的逻辑保持不变，只是延迟实际清理。

### Step 6：安全-1 内联 onclick 替换为事件绑定

定位 `renderInventoryModal` 及其他使用 `innerHTML` + 内联 onclick 的地方：

```js
// 改前
card.innerHTML = `<div onclick="window.equipArmorFromBackpack &&
    window.equipArmorFromBackpack('${entry.id}')">`;

// 改后
const div = document.createElement('div');
div.addEventListener('click', () => {
    if (typeof equipArmorFromBackpack === 'function') {
        equipArmorFromBackpack(entry.id);
    }
});
card.appendChild(div);
```

优先处理涉及 `entry.id` / `entry.key` 来自 localStorage 的场景（XSS 风险最高）。

### Step 7：工具-1 `sync-web.js` 全局变量冲突检测

在合并脚本末尾加检测逻辑：

```js
// sync-web.js 合并后扫描顶层声明
function detectGlobalConflicts(combinedCode) {
    const declarations = new Map();
    const regex = /^(?:var|let|const|function)\s+(\w+)/gm;
    let match;
    while ((match = regex.exec(combinedCode)) !== null) {
        const name = match[1];
        if (declarations.has(name)) {
            console.warn(`[冲突检测] 全局变量重复声明: "${name}"`);
        } else {
            declarations.set(name, true);
        }
    }
}
detectGlobalConflicts(combinedCode);
```
