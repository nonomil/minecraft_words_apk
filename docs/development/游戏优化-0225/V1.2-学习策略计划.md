### Step 2：L-5 挑战触发概率联动掌握度（`12-challenges.js:436`）

定位 `shouldTriggerLearningChallenge`，替换触发逻辑：

```js
function shouldTriggerLearningChallenge(wordObj) {
    if (!settings.learningMode) return false;
    if (!settings.challengeEnabled || currentLearningChallenge) return false;

    const baseFq = Number(settings.challengeFrequency ?? 0.3);
    const seenCount = sessionWordCounts[wordObj.en] || 0;
    const quality = wordPicker?.getWordQuality?.(wordObj.en);

    // 首次见到：降低触发概率，让孩子先充分接触单词
    if (seenCount <= 1) return Math.random() < baseFq * 0.4;

    // 答错过的词：提高触发概率，增加纠错复现
    if (quality === 'wrong') return Math.random() < Math.min(0.85, baseFq * 2);

    // 已熟练掌握：降低触发概率，减少重复
    if (quality === 'correct_fast') return Math.random() < baseFq * 0.6;

    // 默认
    return Math.random() < Math.max(0.1, Math.min(0.9, baseFq));
}
```

### Step 5：G-3 DDA 联动学习表现（`05-difficulty.js` + `12-challenges.js`）

**`12-challenges.js`** — 维护连对/连错计数器：

```js
let sessionCorrectStreak = 0;
let sessionWrongStreak = 0;

// 在 completeLearningChallenge 答对分支：
sessionCorrectStreak++;
sessionWrongStreak = 0;

// 在 completeLearningChallenge 答错分支：
sessionWrongStreak++;
sessionCorrectStreak = 0;

// 暴露给 DDA 模块
function getLearningStreaks() {
    return { correct: sessionCorrectStreak, wrong: sessionWrongStreak };
}
```

**`05-difficulty.js`** — `computeDifficultyState` 末尾追加：

```js
const streaks = getLearningStreaks?.() ?? { correct: 0, wrong: 0 };

// 连续答对 3+：挑战频率临时提升
if (streaks.correct >= 3) {
    settings.challengeFrequency = Math.min(0.9,
        (settings.challengeFrequency ?? 0.3) + 0.08);
}

// 连续答错 2+：优先出最简单题型
if (streaks.wrong >= 2) {
    settings._forceChallengeType = 'translate';
    settings.challengeFrequency = Math.max(0.1,
        (settings.challengeFrequency ?? 0.3) - 0.1);
}

// 连续答错 4+：DDA 降低敌人伤害，给孩子喘息
if (streaks.wrong >= 4) {
    enemyDamageMult *= 0.9;
}

// 连续答对 5+：恢复正常题型
if (streaks.correct >= 5) {
    delete settings._forceChallengeType;
}
```
