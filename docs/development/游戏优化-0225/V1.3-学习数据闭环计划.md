## 二、数据结构设计

### 旧结构（当前）

```js
progress.vocab[packId].unique[word] = 1  // 见过 = 1，无更多信息
```

### 新结构

```js
progress.vocab[packId].unique[word] = {
    seen:     3,              // 见过次数
    correct:  2,              // 答对次数
    wrong:    1,              // 答错次数
    lastSeen: 1708819200000,  // 上次见到时间戳
    quality:  'correct_slow'  // 'new' | 'correct_fast' | 'correct_slow' | 'wrong'
}
```

### 兼容策略

旧存档中 `unique[word] = 1`（数字），新代码读取时检测类型：

```js
function normalizeWordEntry(val) {
    if (typeof val === 'object' && val !== null) return val;  // 已是新结构
    // 旧结构迁移：seen=1，其余字段默认
    return { seen: val || 1, correct: 0, wrong: 0,
             lastSeen: Date.now(), quality: 'new' };
}
```

### Step 2：`normalizeProgress` 加入迁移逻辑（`09-vocab.js`）

在 `normalizeProgress` 函数中，对每个词条做兼容转换：

```js
function normalizeProgress(raw) {
    // ... 现有逻辑 ...

    // 迁移旧词条结构
    if (raw?.vocab) {
        Object.keys(raw.vocab).forEach(packId => {
            const pack = raw.vocab[packId];
            if (pack?.unique) {
                Object.keys(pack.unique).forEach(word => {
                    pack.unique[word] = normalizeWordEntry(pack.unique[word]);
                });
            }
        });
    }
    return raw;
}
```

### Step 3：`recordWordProgress` 写入结构化数据（`12-challenges.js`）

定位 `recordWordProgress`，替换写入逻辑：

```js
function recordWordProgress(wordObj) {
    const en = wordObj.en;
    const pr = getPackProgress();
    if (!pr) return;

    if (!pr.unique[en]) {
        // 首次见到
        pr.unique[en] = { seen: 1, correct: 0, wrong: 0,
                          lastSeen: Date.now(), quality: 'new' };
        pr.uniqueCount++;
    } else {
        // 再次见到
        const entry = normalizeWordEntry(pr.unique[en]);
        entry.seen++;
        entry.lastSeen = Date.now();
        pr.unique[en] = entry;
    }
    saveProgress();
}
```

### Step 5：词库激活时回灌历史 quality（`09-vocab.js`）

在 `setActiveVocabPack` 加载词库后，用历史数据初始化 `wordPicker`：

```js
function setActiveVocabPack(packId) {
    // ... 现有逻辑 ...

    // 回灌历史质量分到 wordPicker
    const pack = progress?.vocab?.[packId];
    if (pack?.unique && wordPicker) {
        Object.entries(pack.unique).forEach(([word, entry]) => {
            const e = normalizeWordEntry(entry);
            if (e.quality && e.quality !== 'new') {
                wordPicker.updateWordQuality(word, e.quality);
            }
        });
    }
}
```

### Step 6：游戏结束界面展示本局词汇（`12-challenges.js` + `Game.html`）

**`12-challenges.js`** — 在 `setOverlay('gameover')` 调用前生成词汇摘要：

```js
function buildSessionWordsSummary() {
    const words = getUniqueSessionWords();  // 已有函数
    if (!words.length) return '';
    const items = words.map(w =>
        `<span class="session-word" onclick="speakWord('${w.en}')"
               title="${w.zh || ''}">${w.en}</span>`
    ).join(' ');
    return `<div class="session-words-summary">
        <div class="session-words-title">本局接触词汇（共 ${words.length} 个）</div>
        <div class="session-words-list">${items}</div>
        <div class="session-words-hint">点击单词可朗读</div>
    </div>`;
}
```

const summaryEl = document.getElementById('overlay-text');
