// 08-account.js
function syncWordStats(en, correct) {
  if (!currentAccount || !en) return;
  currentAccount.vocabulary = currentAccount.vocabulary || {};
  currentAccount.vocabulary.wordStats = currentAccount.vocabulary.wordStats || {};
  const stat = currentAccount.vocabulary.wordStats[en] || { correct: 0, wrong: 0 };
  if (correct) stat.correct += 1;
  else stat.wrong += 1;
  currentAccount.vocabulary.wordStats[en] = stat;
}
window.syncWordStats = syncWordStats;

// 12-challenges.js (completeLearningChallenge 内)
if (typeof syncWordStats === "function" && wordObj?.en) {
  syncWordStats(String(wordObj.en), !!correct);
}
```

// 17-bootstrap.js
async function registerServiceWorkerIfSupported() {
  if (!("serviceWorker" in navigator)) return;
  try {
    await navigator.serviceWorker.register("./service-worker.js");
  } catch (err) {
    console.warn("SW register failed", err);
  }
}
// bootstrap 入口完成后调用
registerServiceWorkerIfSupported();
```
